library IEEE;
use IEEE.STD_LOGIC_1164.ALL;


entity entydades is
    Port ( a,b : in  STD_LOGIC_VECTOR (3 downto 0);
           X,Y,Z : out  STD_LOGIC );
end entydades;

architecture Behavioral of entydades is

begin
process (a,b) begin
 if (a = b) then 
 x <= '1';y <= '0';z <= '0'; 
 elsif (a >  b) then
 x <= '0';y <= '1';z <= '0' ;else
 x <= '0';y <= '0'; z <= '1';
 end if;
 
end process; 
 
end Behavioral;
